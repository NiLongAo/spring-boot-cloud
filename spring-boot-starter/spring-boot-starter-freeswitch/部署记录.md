--privileged=true \
--net=host \
-e TZ=Asia/Shanghai \
-e SOUND_RATES=8000:16000 \

###### 2. /dialplan/default.xml 中 group_dial_sip_proxy 添加 在最底部
```xml
<extension name="group_dial_sip_proxy">        
   <condition field="destination_number" expression="(^\d{4}$|^\d{3}$)">                
       <action application="set"><![CDATA[sip_h_X-Number=<sip:$1@${domain_name}>]]></action>       
       <action application="bridge" data="user/1000@192.168.1.26"/> 
   </condition>
</extension>
```

###### 5.动态配置 fs xml信息 用于动态效应用户等信息
```xml
<!-- 开启模块 mod_xml_curl -->
<!-- 在 xml_curl.conf 中添加 curl 路由，地址为后台配置的地址 bindings 中 -->

<!-- 拨号计划 -->
<binding name="dialplan">
    <param name="gateway-url" value="http://192.168.1.5:8670/api/fs/xml/find_dialplan_xml" bindings="dialplan"/>
</binding>
<!-- 配置用户目录 -->
<binding name="directory">
    <param name="gateway-url" value="http://192.168.1.5:8670/api/fs/xml/find_directory_xml" bindings="directory"/>
</binding>
<!-- 配置项 -->
<binding name="configuration">
    <param name="gateway-url" value="http://192.168.1.5:8670/api/fs/xml/find_configuration_xml" bindings="configuration"/>
</binding>
```

###### 5.中文 设置
```xml
    <!--    1.修改vars中的中文语言配置 可将英文音频资源关闭 修改为中文音频资源的路径设置-->
    <X-NO-PRE-PROCESS cmd="set" data="sound_prefix=$${sounds_dir}/en/us/callie"/> 
    <X-PRE-PROCESS cmd="set" data="sound_prefix=$${sounds_dir}/zh/cn/link"/> 
    <!--  2.增加vars中的字段如下 设置默认的语言 设置默认的方言 设置默认的声音 -->
    <X-PRE-PROCESS cmd="set" data="default_language=zh"/>   
    <X-PRE-PROCESS cmd="set" data="default_dialect=cn"/>     
    <X-PRE-PROCESS cmd="set" data="default_voice=link"/>  
    <!--  
        cd lang
        cp -fr en zh
        cd zh
        mv en.xml zh.xml
     -->
    <!-- 修改zh.xml文件 -->
    <language name="zh" say-module="zh" sound-prefix="$${sounds_dir}/zh/cn/link" tts-engine="mod_tts_commandline" tts-voice="Ting-Ting">
    <!--  编辑conf/autoload_configs/modules.conf.xml 默认注掉的，取消注释即可 -->
    <load module="mod_say_zh"/>
    <!--    添加配置文件freeswitch.xml中文语言信息-->
    <section name="languages" description="Language Management">
        <X-PRE-PROCESS cmd="include" data="lang/de/*.xml"/>
        <X-PRE-PROCESS cmd="include" data="lang/en/*.xml"/>
        <X-PRE-PROCESS cmd="include" data="lang/zh/*.xml"/>


```
